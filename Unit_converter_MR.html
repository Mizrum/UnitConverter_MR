print "Welcome to our Kilometer to Miles converter." 
 
 
kilometers = int(raw_input("To get the Miles, please enter number of kilometers: ")) 
 
 
while True: 
    try: 
        kilometers = float(kilometers.replace(",", ".")) 
        miles = kilometers * 0.621371 
        print float(miles) 
        break 
    except ValueError: 
        print "Please enter only a number!"
